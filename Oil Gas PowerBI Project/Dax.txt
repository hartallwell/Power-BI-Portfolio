Executive Dashboard Title =
VAR SelectedYears = VALUES(Dim_Date[Year])
VAR CountYears = COUNTROWS(SelectedYears)
VAR TotalYears = CALCULATE(DISTINCTCOUNT(Dim_Date[Year]), ALL(Dim_Date))

RETURN
SWITCH(
    TRUE(),
    CountYears = TotalYears,
        "Executive Dashboard - All Years",
    CountYears = 1,
        "Executive Dashboard - " &
        SELECTEDVALUE(Dim_Date[Year]),
    "Executive Dashboard - " &
        CONCATENATEX(
            SelectedYears,
            Dim_Date[Year],
            ", ",
            Dim_Date[Year],
            ASC
        )
)



Highest Cost Facility =
VAR TopFacility =
    TOPN(
        1,
        SUMMARIZE(
            Fact_Maintenance,
            Dim_Facility[FacilityName],
            "TotalCost", SUM(Fact_Maintenance[Cost])
        ),
        [TotalCost], DESC
    )
RETURN
"The highest maintenance cost was recorded at "
    & MAXX(TopFacility, Dim_Facility[FacilityName])
    & "."


Highest Downtime Equipment =
VAR TopEquipment =
    TOPN(
        1,
        SUMMARIZE(
            Fact_Maintenance,
            Dim_Equipment[EquipmentName],
            "TotalDowntime", SUM(Fact_Maintenance[DowntimeHours])
        ),
        [TotalDowntime], DESC
    )
RETURN
MAXX(TopEquipment, Dim_Equipment[EquipmentName])
    & " accounted for the highest downtime during the selected period."

Report Information =
VAR StartDate =
    MIN(Dim_Date[Date])

VAR EndDate =
    MAX(Dim_Date[Date])

VAR RefreshDate =
    MAX('Last Refresh'[Last Refresh])

RETURN
"📅 Data Period: "
& FORMAT(StartDate, "MMMM yyyy")
& " - "
& FORMAT(EndDate, "MMMM yyyy")
& "    |    🔄 Last Refresh: "
& FORMAT(RefreshDate, "dd MMM yyyy HH:mm")
